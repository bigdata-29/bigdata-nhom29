
Chạy bên ngoài:
1. sudo apt-get install build-essential libssl-dev libffi-dev python3-dev python3-pip libsasl2-dev libldap2-dev default-libmysqlclient-dev


Tạo môi trường ảo 
python3 -m venv superset_env
. superset_env/bin/activate
pip install --upgrade setuptools pip


Các gói bắt buộc:
pip install pillow apache-superset


export SUPERSET_CONFIG_PATH=$PWD/superset/superset_config.py

khởi tạo superset (init)
export FLASK_APP=superset

superset db upgrade

superset fab create-admin

superset init


Tạo file run_superset.sh
Nâng quyền chmod +x run_superset.sh

Chạy thử sh run_superset.sh và test trong localhost:8088

Tạo thành service superset
sudo nano /etc/systemd/system/superset.service

[Unit]
Description = Apache Superset Webserver Daemon
After = network.target

[Service]
PIDFile = /home/hoanvdtd/superset/superset-webserver.PIDFile
Environment=SUPERSET_HOME=/home/hoanvdtd/superset
Environment=PYTHONPATH=/home/hoanvdtd/superset
WorkingDirectory = /home/hoanvdtd/superset
limit-re>
ExecStart = /home/hoanvdtd/superset/run_superset.sh
ExecStop = /bin/kill -s TERM $MAINPID


[Install]
WantedBy=multi-user.target

Chạy service dưới dạng local app
sudo systemctl daemon-reload
sudo systemctl enable superset.service
sudo systemctl start superset.service
